#include <stdio.h>
#include <stdio.h>

//There are no syntax errors in this code
//But it has an example of shift reduce conflict. 

void print(int x)
{
	int i=0,j=0;
// The following statement can produce shift reduce conflict if no precedence of grammars is given
	int y = 5+x*10;

	while(i<x)
	{
		j=0;
		while(j<y)
		{
			if(i==j)
			{
				printf("equal");
			}
			else if (i>j)
			{
				printf("Greater");
			}
			else
			{
				printf("Smaller");
			}
			j++;
		}
		printf("\n");
		i++;
	}
}

int main()
{
	print(5);
	return 0;
}
