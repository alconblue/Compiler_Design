#include <stdio.h>
#include <stdio.h>

int print(int x, int y)
{
	int i=0, j=0;

	while(i<x)
	{
		j=0;
		while(j<y)
		{
			if(i==j)
			{
				printf("equal");
			}
			
			j++;
		}
		printf("\n");
		i++;
	}
}
